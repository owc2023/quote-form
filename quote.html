  <script>

    function calculateQuote() {

      const bedrooms = parseInt(document.getElementById('bedrooms').value);

      const stories = parseInt(document.getElementById('stories').value);

      const condition = parseFloat(document.getElementById('condition').value);

      const service = parseFloat(document.getElementById('service').value);


      let base = bedrooms * 40;

      if (stories === 2) base += 30;

      if (stories === 3) base += 60;


      const finalPrice = (base * condition * service).toFixed(2);


      document.getElementById('result').style.display = 'block';

      document.getElementById('result').innerHTML = `Estimated Quote: <strong>$${finalPrice}</strong>`;

    }

  </script>
